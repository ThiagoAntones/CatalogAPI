			AND `Extra` = 'auto_increment'
			AND `COLUMN_KEY` = 'PRI'

		SELECT `COLUMN_TYPE`

			FROM `information_schema`.`COLUMNS`

				AND `TABLE_NAME` = TABLE_NAME_ARGUMENT
				AND `COLUMN_KEY` = 'PRI'
			LIMIT 1;
		SELECT `COLUMN_NAME`
			INTO PRIMARY_KEY_COLUMN_NAME
			FROM `information_schema`.`COLUMNS`
			WHERE `TABLE_SCHEMA` = (SELECT IFNULL(SCHEMA_NAME_ARGUMENT, SCHEMA()))
